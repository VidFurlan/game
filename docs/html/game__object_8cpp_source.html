111 float rotation = mPos.z;
112 const GameObject *parent = pParent;
113
114 while (parent) {
115 float parentRotation = parent->mPos.z;
116 glm::vec2 parentPos(parent->mPos.x, parent->mPos.y);
117
118 float radians = glm::radians(parentRotation);
119
120 float cosAngle = cos(radians);
121 float sinAngle = sin(radians);
122 pos = glm::vec2(
123 cosAngle * pos.x - sinAngle * pos.y,
